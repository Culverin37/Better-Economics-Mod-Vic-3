		law_mass_conscription = strongly_approve
		law_professional_army = approve
		law_national_militia = strongly_disapprove
		law_peasant_levies = strongly_disapprove
	}

	lawgroup_free_speech = {
		law_protected_speech = strongly_disapprove
		law_right_of_assembly = disapprove
		law_censorship = neutral

	}

	lawgroup_policing = {

	}

		law_pec_persecuted_unions = disapprove
		law_pec_regulated_unions = approve
		law_pec_free_unions = strongly_disapprove
	}

	lawgroup_pec_state_autonomy = {
		law_pec_nobility = strongly_disapprove
		law_pec_unitarism = approve
		law_pec_federalism = disapprove
		law_pec_confederation = strongly_disapprove
	}
	
	possible = {
		owner = { has_technology_researched = mass_propaganda }
		# Agitator ideologies should disapprove of a current law
		NAND = {
			has_role = agitator
			owner = {
				has_law = law_type:law_autocracy
				has_law = law_type:law_ethnostate
				has_law = law_type:law_outlawed_dissent
				has_law = law_type:law_militarized_police
			}
		}
		# Any Interest Group
		scope:interest_group = {
			NOR = {
				has_variable = communist_ig_var
				has_variable = chose_vanguardism
				has_variable = chose_ethno_nationalism
			}
		}
	}
	
	leader_weight = {
		value = 100

		# more likely in recession
		if = {
			limit = {
				owner = {
					has_modifier = pec_boombust_recession
				}
			}
			add = 50
		}

		# more likely in depression
		if = {
			limit = {
				owner = {
					has_modifier = pec_boombust_depression
				}
			}
			add = 100
		}

		# more likely for old world intelligentsia
		if = {
			limit = {
				scope:interest_group = {
					has_ideology = ideology:ideology_pec_oldworld_liberal
				}
			}
			add = 25
		}

		# much more likely for ethnically reformed old world intelligentsia
		if = {
			limit = {
				scope:interest_group = {
					has_ideology = ideology:ideology_pec_oldworld_ethnic
				}
			}
			add = 75
		}

		# Less likely with two-party system
		if = {
			limit = {
				OR = {
					owner = { has_modifier = pec_two_party_system }
					owner = { has_modifier = pec_two_party_system_socialist }
				}
			}
			multiply = { value = 0.33 }
		}

		# More likely if paying war reparations
		if = {
			limit = {
				any_country = {
					has_diplomatic_pact = {
						who = root.owner
						type = war_reparations
						is_initiator = yes
					}
				}
			}
			add = 150
		}
		# More likely with a convenient scapegoat
		if = {
			limit = {
				owner = {
					any_scope_state = {
						is_incorporated = yes
						any_scope_pop = {
							pop_is_discriminated = yes
						}
					}
				}
			}
			add = 50
		}
		# more likely if bankrupt
		if = {
			limit = {
				owner = {
					in_default = yes
				}
			}
			add = 50
		}

		# Less likely in council republic
		if = {
			limit = {
				owner = {
					has_law = law_type:law_council_republic
				}
			}
			add = -75
		}
		# more likely if angry non-unions
		if = {
			limit = {
				owner = {
					NOT = { has_law = law_type:law_ethnostate }
				}
				scope:interest_group = {
					ig_approval <= -5
					NOT = { is_interest_group_type = ig_trade_unions }
				}
			}
			add = 75
		}
		# less likely if content without ethnostate
		if = {
			limit = {
				owner = {
					NOT = {
						has_law = law_type:law_ethnostate
					}
				}
				scope:interest_group = {
					ig_approval >= 5
				}
			}
			add = -75
		}
		# more likely for PB and armed forces
		if = {
			limit = {
				scope:interest_group = {
					is_interest_group_type = ig_petty_bourgeoisie
					is_interest_group_type = ig_armed_forces
				}
			}
			add = 50
		}
		# less likely for unions and intelligentsia
		if = {
			limit = {
				scope:interest_group = {
					is_interest_group_type = ig_trade_unions
					is_interest_group_type = ig_intelligentsia
				}
			}
			add = -25
		}
		# more likely if in government with ethnostate
		if = {
			limit = {
				owner = {
					has_law = law_type:law_ethnostate
					NOR = {
						is_enacting_law = law_type:law_multicultural
						is_enacting_law = law_type:law_national_supremacy
						is_enacting_law = law_type:law_racial_segregation
						is_enacting_law = law_type:law_cultural_exclusion
					}
				}
				scope:interest_group = {
					is_in_government = yes
				}
			}
			add = 150
		}
		# conservative and/or reactionary leaders have a higher spawn chance if the Anti-Dreyfusards won.
		if = {
			limit = {
				scope:interest_group = {
					has_variable = anti_dreyfusard_victory
				}
			}
			add = 50
		}
		# red scare can make you more fascist
		if = {
			limit = {
				scope:interest_group = {
					has_variable = fascist_uptick
				}
			}
			add = 25
		}
		if = {
			limit = {
				OR = {
					owner = { has_modifier = pec_two_party_system }
					owner = { has_modifier = pec_two_party_system_socialist }
				}
			}
			multiply = { value = 0.33 }
		}
		if = { #Brazil has Integralists instead
			limit = {
				owner = {
					c:BRZ ?= this
				}
			}
			multiply = 0
		}
	}
}
