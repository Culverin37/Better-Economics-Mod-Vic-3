law_pec_nobility = {
	group = lawgroup_pec_state_autonomy
	
	icon = "gfx/interface/icons/law_icons/pec_nobility.dds"
	
	progressiveness = -75

	disallowing_laws = {
	}
	
	on_enact = {
		if = {
			limit = {
				has_law = law_type:law_pec_unitarism
			}
			custom_tooltip = pec_unitarism_removed_tt
		}
		if = {
			limit = {
				has_law = law_type:law_pec_confederation
			}
			custom_tooltip = pec_confederation_removed_tt
		}
	}

	unlocking_technologies = {
	}
	
	modifier = {
		country_aristocrats_pol_str_mult = 0.25
		state_market_access_price_impact = -0.075
		state_bureaucracy_population_base_cost_factor_mult = -0.15
		state_tax_capacity_mult = -0.10
		country_revolution_progress_mult = 0.25
		country_secession_progress_mult = 0.25
	}
	
	possible_political_movements = {
		law_pec_federalism
		law_pec_unitarism
		law_pec_confederation
	}

	pop_support = {
		value = 0
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
	}
}

law_pec_unitarism = {
	group = lawgroup_pec_state_autonomy
	
	icon = "gfx/interface/icons/law_icons/unused/home_affairs.dds"
	
	progressiveness = -25

	disallowing_laws = {
	}
	
	on_enact = {
		if = {
			limit = {
				OR = {
					has_law = law_type:law_pec_nobility
					has_law = law_type:law_pec_federalism
					has_law = law_type:law_pec_confederation
				}
			}
			custom_tooltip = pec_unitarism_enacted_tt
		}
		if = {
			limit = {
				has_law = law_type:law_pec_confederation
			}
			custom_tooltip = pec_confederation_removed_tt
		}
	}

	unlocking_technologies = {
		tech_bureaucracy
	}
	
	modifier = {
		country_authority_add = 100
		country_bureaucrats_pol_str_mult = 0.33
		state_market_access_price_impact = 0.02
	}
	
	possible_political_movements = {
		law_pec_federalism
	}

	pop_support = {
		value = 0
		add = {
			desc = "POP_CAPITAL"			
			if = {
				limit = {
					state = {
						is_capital = yes
					}
				}
				value = 0.25
			}
		}
		add = {
			desc = "POP_CAPITAL_LATINO"			
			if = {
				limit = {
					state = {
						is_capital = yes
					}
					OR = {
						owner = { country_has_primary_culture = cu:platinean }
						owner = { country_has_primary_culture = cu:south_andean }
						owner = { country_has_primary_culture = cu:north_andean }
						owner = { country_has_primary_culture = cu:central_american }
						owner = { country_has_primary_culture = cu:mexican }
						owner = { country_has_primary_culture = cu:brazilian }
					}
				}
				value = 0.50
			}
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
	}
}
