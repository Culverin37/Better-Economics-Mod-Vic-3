		video = "asia_poor_people_moving"
	}

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	trigger = {

	}

	immediate = {
		random_state = {
			limit = {
				state_region = s:STATE_OUTER_MANCHURIA
			}
			save_scope_as = outer_manchuria_scope
		}
		c:CHI = {
			save_scope_as = china_scope
			create_truce = {
				country = c:RUS
				months = normal_modifier_time
			}
		}
	}

	option = {
		name = russo_chinese.1.a
		default_option = yes
		if = {
			limit = {
				c:CHI = {
					owes_obligation_to = c:RUS
				}
			}
			c:CHI = {
				set_owes_obligation_to = {
					country = c:RUS
					setting = no
				}
			}		
		}
		random_state = {
			limit = {
				state_region = s:STATE_OUTER_MANCHURIA
			}
			set_state_owner = c:RUS
 			add_modifier = {
 				name = modifier_ruler_of_the_east
 				months = long_modifier_time
 			}
		}
	}

	option = {
		name = russo_chinese.1.b
		if = {
			limit = {
				c:CHI = {
					owes_obligation_to = c:RUS
				}
			}
			c:CHI = {
				set_owes_obligation_to = {
					country = c:RUS
					setting = no
				}
			}		
		}
		random_state = {
			limit = {
				state_region = s:STATE_OUTER_MANCHURIA
			}
