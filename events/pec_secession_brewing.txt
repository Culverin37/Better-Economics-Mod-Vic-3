pec_secession_brewing.1 = {
	type = country_event
	hidden = yes
	

	trigger = {
		NOT = { has_game_rule = pec_nationalism_impossible }
		has_technology_researched = nationalism
		highest_secession_progress = 0
		NOR = {
			has_modifier = pec_suppressed_brewing_nationalists
			has_variable = pec_completed_brewing_secession_je
			has_journal_entry = pec_je_secessionist_movement
			is_revolutionary = yes
			is_secessionist = yes
		}
		any_scope_state = {
			OR = {
				is_incorporated = yes
				AND = {
					pec_state_is_in_europe = yes
					owner = {
						pec_country_is_in_europe = yes
					}
				}
				AND = {
					state_is_in_east_asia = yes
					owner = {
						country_is_in_east_asia = yes
					}
				}
				AND = {
					state_is_in_north_america = yes
					owner = {
						country_is_in_north_america = yes
					}
				}
				AND = {
					state_is_in_central_america = yes
					owner = {
						country_is_in_central_america = yes
					}
				}
				AND = {
					state_is_in_south_america = yes
					owner = {
						country_is_in_south_america = yes
					}
				}
				AND = {
					state_is_in_africa = yes
					owner = {
						country_is_in_africa = yes
					}
				}
				AND = {
					state_is_in_southeast_asia = yes
					owner = {
						country_is_in_southeast_asia = yes
					}
				}
			}
			turmoil > 0.15
			state_population > 1000000
			NOR = {
				has_modifier = pec_nationalism_crushed_state
				has_modifier = pec_concessions_accepted_state
				has_modifier = pec_isolated_nationalist_home_rule
			}
			OR = {
				NOT = { is_homeland_of_country_cultures = ROOT }
				has_game_rule = pec_nationalism_relaxed_req
			}
			save_temporary_scope_as = pec_potential_secessionist_state
			any_scope_culture = {
				NOT = { is_primary_culture_of = ROOT }
				has_homeland = scope:pec_potential_secessionist_state
				save_temporary_scope_as = pec_potential_secessionist_culture
			}
			culture_percent_state = {
				target = scope:pec_potential_secessionist_culture
				value >= 0.5
			}
		}
	}
	immediate = {
		random_list = {
			1 = {
				modifier = {
					if = {
						limit = {
							has_journal_entry = je_springtime_of_the_peoples
						}
					}
					add = 8
				}
				modifier = {
					if = {
						limit = {
							has_law = law_type:law_pec_unitarism
						}
					}
					add = 1
				}
				modifier = {
					if = {
						limit = {
							government_legitimacy < 75
						}
					}
					add = 1
				}
				modifier = {
					if = {
						limit = {
							government_legitimacy < 50
						}
					}
					add = 1
				}
				modifier = {
					if = {
						limit = {
							government_legitimacy < 25
						}
					}
					add = 1
				}
				add_journal_entry = {
					type = pec_je_secessionist_movement
				}
			}
			9 = {
				modifier = {
					if = {
						limit = {
							has_law = law_type:law_pec_federalism
						}
					}
					add = 10
				}
			}
		}
	}
}
