namespace = bem_workhouse_pm_change

bem_workhouse_pm_change.1 = {
    type = country_event
    event_image = {
	}
	on_created_soundeffect = ""
	icon = ""
    #placement = ROOT
    #THIS = ROOT
    title = bem_workhouse_pm_change.1.t
	desc = bem_workhouse_pm_change.1.d
	flavor = bem_workhouse_pm_change.1.f
	duration = 3
    immediate = {
        every_scope_building = {
            limit = {
                is_building_type = building_workhouses 
                not = { has_active_production_method = bem_pm_workhouse_vhmort }
            }
            activate_production_method = {bem_pm_workhouse_hmort}
        }
    }
    option = {
		name = bem_workhouse_pm_change.1.a
		default_option = yes
	}
}
