namespace = bem_workhouse_pm_change

bem_workhouse_pm_change.1 = {
    type = country_event
    placement = ROOT
    THIS = ROOT
    immediate = {
        every_scope_building = {
            limit = {
                is_building_type = building_workhouses 
                not = { has_active_production_method = bem_pm_workhouse_vhmort }
            }
        }
        activate_production_method = {bem_pm_workhouse_vhmort}
    }
}
